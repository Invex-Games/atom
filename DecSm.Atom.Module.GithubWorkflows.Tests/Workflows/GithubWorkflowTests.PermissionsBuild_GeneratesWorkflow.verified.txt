name: permissions-workflow

on:
  pull_request:
    branches: [ main ]

permissions: read-all

jobs:

  PermissionsTarget:
    permissions:
      actions: write
    runs-on: ubuntu-latest
    steps:

      - name: Checkout
        uses: actions/checkout@v6
        with:
          fetch-depth: 0

      - name: PermissionsTarget
        id: PermissionsTarget
        run: dotnet run --project AtomTest/AtomTest.csproj -- PermissionsTarget --skip --headless
